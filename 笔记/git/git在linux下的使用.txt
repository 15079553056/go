//-----------------------------------------//
查看版本号

	git --version
  
//-----------------------------------------//
安装 git

	sudo apt install git

//-----------------------------------------//
配置 git 账号

	git config --global user.name "git的用户名"

    git config --global user.email "git的邮箱"
	
//-----------------------------------------//
克隆GitHub上面仓库(使用https)

	git clone 仓库路径

//-----------------------------------------//
将文件添加到Git版本控制

	git add 文件名
	
//-----------------------------------------//
查看状态

	git status             查看代码在缓存与当前工作目录的状态
	git status -s          将结果以简短的形式输出
	
//-----------------------------------------//
git提交到本地仓库

	git commit -m "提交的信息"
	
	如果是有增加新文件
	
	git commit -am "提交的信息"
	
//-----------------------------------------//
将提交的内容同步到 github 上

	git push -u origin 分支名

//-----------------------------------------//
查看日志

	git log
